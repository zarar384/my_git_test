-- RELATIONS
1:1
- embed:
user { profile: {} }
- ref:
user { profileId }

1:N
- embed (small):
user { orders: [] }
- ref (normal):
order { userId }

M:N
- arrays:
student { courseIds: [] }
- join (better):
enrollment { studentId, courseId }

RULE:
small -> embed
big -> reference

- cursor keeps position
- no var -> new query every time

-- SORT

multi sort:

-- SKIP (pagination)

-- LIMIT

-- SKIP + LIMIT
db.movies.find().skip(10).limit(10)

page 2 (10 per page)

-- ORDER (Mongo handles it)
.find().sort().skip().limit()

real execution:
sort -> skip -> limit

-- IMPORTANT
cursor = stream, NOT array
data NOT loaded all at once

next() -> one by one
forEach() -> all

-- ERROR
cursor.next() -> error if empty

fix:
cursor.hasNext()

-- BEST PRACTICE

BAD:
find().forEach(doc => if (doc.rating > 9))

GOOD:
find({ "meta.rating": { $gt: 9 } })

ALWAYS filter in DB, not client!

-- WRITE CONCERN & JOURNALING

-- WHAT IS THIS:
writeConcern = how sure MongoDB is that your data is actually saved

-- PARAMS:
w -> how many nodes confirm
j -> write to journal (disk) or not

-- SAFE (don’t lose data)
db.companies.insertOne(
{
  name: "SafeCompany"
},
{
  writeConcern:
  {
    w: "majority",
    j: true
  }
})

-- result:
-- survives crash
-- data recoverable
-- slower

-- FAST (can lose data)
db.companies.insertOne(
{
  name: "FastCompany"
},
{
  writeConcern:
  {
    w: 1,
    j: false
  }
})

-- result:
-- fast
-- may lose data on crash

-- DEFAULT:
{ w: 1, j: false }

-- WHEN TO USE:
-- j: true => safe
-- important data (payments, users)

-- j: false => risk
-- logs, cache, temp data
